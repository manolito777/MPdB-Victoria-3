je_russo_turkish_war = {
	icon = "gfx/interface/icons/event_icons/event_military.dds"

	group = je_group_russia_expansion

	is_shown_when_inactive = {
		exists = c:RUS
        THIS = c:RUS
        is_subject = no
	}
	possible = {
		has_technology_researched = nationalism
		any_interest_group = {
			is_powerful = yes
			is_in_government = yes
			leader = {
				OR = {
					has_ideology = ideology:ideology_jingoist_leader
					has_ideology = ideology:ideology_ethno_nationalist
					has_ideology = ideology:ideology_fascist
				}
			}
		}
		custom_tooltip = {
			text = caucasus_secured_tt
			has_global_variable = caucasus_consolidated_var
		}
	}

	immediate = {
		trigger_event = { id = russo-turkish_war.1  popup = yes }

		set_variable = russo-turkish_war_initialization
	}

	complete = {
		has_variable = russo-turkish_war_initialization
	}

	on_complete = {
		
	}

	fail = {

	}

	on_fail = {

	}

	should_be_pinned_by_default = yes

	weight = 100
}
