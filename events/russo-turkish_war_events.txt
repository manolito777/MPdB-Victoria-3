russo-turkish_war.1 = { # start event
	type = country_event
	placement = root
	title = russo-turkish_war.1.t
	desc = russo-turkish_war.1.d
	flavor = russo-turkish_war.1.f

	duration = 3

	event_image = {
		video = "unspecific_signed_contract"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_map.dds"

	trigger = {
	}

	immediate = {
		random_interest_group = {
			limit = {
				is_powerful = yes
				is_in_government = yes
				OR = {
					is_interest_group_type = ig_armed_forces
					leader = {
						OR = {
							has_ideology = ideology:ideology_jingoist_leader
							has_ideology = ideology:ideology_ethno_nationalist
							has_ideology = ideology:ideology_fascist
						}
					}
				}
			}
			leader = { save_scope_as = borders_leader } #Esto se usa en la localización para nombrar de manera dinámica y dar detalle así.
		}
		c:RUS = { save_scope_as = russo_scope } #Idem.
	}

	option = {
		name = russo-turkish_war.1.a
		default_option = yes
		ai_chance = {
			base = 50
		}
		
		add_modifier = {
			name = russian_big_border_pursuit
			days = very_long_modifier_time
			# country_infamy_decay_mult
		}

		add_journal_entry = {
			type = je_great_russo_turkish_war
		}
		every_country = {
			limit = {
				NOT = { THIS = ROOT }
				any_scope_state = {
					OR = {
						state_region = s:STATE_WALLACHIA
						state_region = s:STATE_BULGARIA
						state_region = s:STATE_DOBRUDJA
					}
				}
			}
			trigger_event = { id = russo-turkish_war.5 } #Respuesta del jugador Otomano. Mejorar relaciones con potencias occidentales y/o conscriptos en las regiones afectadas.
		}
		hidden_effect = {
			every_country = {
				limit = {
					has_diplomatic_relevance = c:RUS
					NOT = { THIS = ROOT }
				}
				post_notification = russo_big_border_notification
			}
		}
	}

	option = { #Western War
		name = russo-turkish_war.1.b

		ai_chance = {
			base = 20
		}

		add_journal_entry = {
			type = je_western_russo_turkish_war
		}

		add_modifier = {
			name = russian_small_border_pursuit
			days = very_long_modifier_time
			# country_infamy_decay_mult
		}

		every_country = {
			limit = {
				NOT = { THIS = ROOT }
				any_scope_state = {
					OR = {
						state_region = s:STATE_WALLACHIA
						state_region = s:STATE_BULGARIA
						state_region = s:STATE_DOBRUDJA
					}
				}
			}
			trigger_event = { id = russo-turkish_war.5 } #Respuesta del jugador Otomano. Mejorar relaciones con potencias occidentales y/o conscriptos en las regiones afectadas.
		}

		hidden_effect = {
			every_country = {
				limit = {
					has_diplomatic_relevance = c:RUS
					NOT = { THIS = ROOT }
				}
				post_notification = russo_small_border_notification
			}
		}
	}

	option = { #Eastern War
		name = russo-turkish_war.1.c

		ai_chance = {
			base = 20
		}

		every_country = {
			limit = {
				NOT = { THIS = ROOT }
				any_scope_state = {
					OR = {
						state_region = s:STATE_KARS
						state_region = s:STATE_TRABZON
						state_region = s:STATE_ERZURUM
					}
				}
			}
			trigger_event = { id = russo-turkish_war.5 } #Respuesta del jugador Otomano. Mejorar relaciones con potencias occidentales y/o conscriptos en las regiones afectadas.
		}
		
		add_journal_entry = {
			type = je_eastern_russo_turkish_war
		}

		add_modifier = {
			name = russian_small_border_pursuit
			days = very_long_modifier_time
			# country_infamy_decay_mult
		}


		hidden_effect = {
			every_country = {
				limit = {
					has_diplomatic_relevance = c:RUS
					NOT = { THIS = ROOT }
				}
				post_notification = russo_small_border_notification
			}
		}

	}

	option = { # imperialist IG's unhappy
		name = russo-turkish_war.1.d
		ai_chance = {
			base = 10
		}
		custom_tooltip = {
			text = dismiss_russo-turkish_war_tt
			set_variable = reject_russo-turkish_war
		}
		every_interest_group = {
			limit = {
				is_powerful = yes
				is_in_government = yes
				OR = {
					is_interest_group_type = ig_armed_forces
					leader = {
						OR = {
							has_ideology = ideology:ideology_jingoist_leader
							has_ideology = ideology:ideology_ethno_nationalist
							has_ideology = ideology:ideology_fascist
						}
					}
				}
			}
			add_modifier = {
				name = no_russo-turkish_war
				days = long_modifier_time
				is_decaying = yes
				# interest_group_approval_add
			}
		}

		add_modifier = {
			name = russian_gentle_border_pursuit
			years = 5
			# country_infamy_decay_mult
		}
	}
}

russo-turkish_war.5 = { #Respuesta del Otomano
	type = country_event
	placement = scope:russian_claim_state
	title = russo-turkish_war.5.t
	desc = russo-turkish_war.5.d

	duration = 3

	dlc = dlc004

	event_image = {
		video = "europenorthamerica_before_the_battle"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_map.dds"

	trigger = {
		# triggered by russo-turkish_war.1
	}

	immediate = {
		c:RUS = {
			save_scope_as = russian_turkwar_scope
		}
		random_scope_state = {
			limit = {
				OR = {
					state_region = s:STATE_WALLACHIA
					state_region = s:STATE_BULGARIA
					state_region = s:STATE_DOBRUDJA
					state_region = s:STATE_KARS
					state_region = s:STATE_TRABZON
					state_region = s:STATE_ERZURUM
				}
			}
			save_scope_as = russian_claim_state
		}
	}

	option = { # Prepare for war
		name = russo-turkish_war.5.a
		default_option = yes
		every_scope_state = {
			limit = {
				OR = {
					state_region = s:STATE_WALLACHIA
					state_region = s:STATE_BULGARIA
					state_region = s:STATE_DOBRUDJA
					state_region = s:STATE_KARS
					state_region = s:STATE_TRABZON
					state_region = s:STATE_ERZURUM
				}
			}
			add_modifier = {
				name = rally_against_russia
				years = 10
				# state_conscription_rate_mult
			}
		}
		ai_chance = {
			base = 15
		}
	}

	option = { # Try to appease France
		name = russo-turkish_war.5.b
		change_relations = {
			country = c:RUS
			value = 50
		}

		create_bidirectional_truce = { country = scope:russian_turkwar_scope months = 12 }

		ai_chance = {
			base = 5
		}
	}
}
